document.getElementById('loginForm').addEventListener('submit', function(e) { e.preventDefault(); let username = document.getElementById('username').value; let password = document.getElementById('password').value; let resultDiv = document.getElementById('result'); let storedUser = localStorage.getItem('user-' + username); if (storedUser) { let user = JSON.parse(storedUser); if (user.password === password) { resultDiv.textContent = 'Login successful. Welcome ' + username + '!'; resultDiv.style.color = 'green'; // Save session state localStorage.setItem('sessionUser', username); // Redirect to dashboard window.location.href = 'dashboard.html'; } else { resultDiv.textContent = 'Incorrect password!'; resultDiv.style.color = 'red'; } } else { resultDiv.textContent = 'User not found!'; resultDiv.style.color = 'red'; } });
